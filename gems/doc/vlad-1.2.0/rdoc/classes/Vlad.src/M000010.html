  def self.load options = {}
    options = {:config => options} if String === options

    recipes = {
      :app    => :mongrel,
      :config => 'config/deploy.rb',
      :core   => :core,
      :scm    => :subversion,
      :web    => :apache,
    }.merge(options)

    recipes.each do |flavor, recipe|
      next if recipe.nil? or flavor == :config
      require "vlad/#{recipe}"
    end

    Kernel.load recipes[:config]
  end
