  def export(revision_or_source, destination)
    if revision_or_source =~ /^(\d+|head)$/i then
      "#{svn_cmd} export -r #{revision_or_source} #{repository} #{destination}"
    else
      "#{svn_cmd} export #{revision_or_source} #{destination}"
    end
  end
