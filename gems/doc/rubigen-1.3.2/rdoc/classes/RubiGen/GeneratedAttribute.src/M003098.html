    def default
      @default ||= case type
        when :integer                     then 1
        when :float                       then 1.5
        when :decimal                     then "9.99"
        when :datetime, :timestamp, :time then Time.now.to_s(:db)
        when :date                        then Date.today.to_s(:db)
        when :string                      then "MyString"
        when :text                        then "MyText"
        when :boolean                     then false
        else
          ""
      end      
    end
