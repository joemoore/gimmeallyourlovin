    def assert_generated_module(path)
      path=~/\/?(\w+)$/
      module_name=$1.camelize
      assert_generated_file("#{path}.rb") do |body|
        assert body=~/module #{module_name}/,"the file '#{path}.rb' should be a module"
        yield body if block_given?
      end
    end
