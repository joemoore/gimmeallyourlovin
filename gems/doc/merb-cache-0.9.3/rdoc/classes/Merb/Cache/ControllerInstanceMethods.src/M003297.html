  def expire(options)
    Merb::Controller._cache.expire_key_for(options, controller_name) do |key, match|
      if match
        Merb::Controller._cache.store.expire_match(key)
      else
        Merb::Controller._cache.store.expire(key)
      end
    end
    true
  end
