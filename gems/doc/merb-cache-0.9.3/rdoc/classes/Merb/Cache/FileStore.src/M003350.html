  def expire_match(key)
    #files = Dir.glob(@config[:cache_directory] / "#{key}*.cache")
    files = Dir.glob(@config[:cache_directory] / "#{key}*")
    FileUtils.rm_rf(files)
    Merb.logger.info("cache: expired matching (#{key})")
    true
  end
