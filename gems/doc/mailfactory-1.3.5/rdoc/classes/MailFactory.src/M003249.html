  def quote_address_if_necessary(address, charset)
    if Array === address
      address.map { |a| quote_address_if_necessary(a, charset) }
    elsif address =~ /^(\S.*)\s+(<.*>)$/
      address = $2
      phrase = quote_if_necessary($1.gsub(/^['"](.*)['"]$/, '\1'), charset, true)
      "\"#{phrase}\" #{address}"
    else
      address
    end
  end
