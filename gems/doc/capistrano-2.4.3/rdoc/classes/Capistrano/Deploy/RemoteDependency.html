57:         @success = true
58:

60:         output_per_server.each_pair do |server, output| 
61:           next if output =~ expect
62:           errored_hosts << server 
63:         end 
64:

66:           @hosts = errored_hosts.join(', ')
67:           output = output_per_server[errored_hosts.first]
68:           @message = "the output #{output.inspect} from #{command.inspect} did not match #{expect.inspect}"
69:           @success = false
70:         end
